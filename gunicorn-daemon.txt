[Unit]
Description=gunicorn daemon
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/var/www/server/Data_retrieving_ecommerce
ExecStart=(which gunicorn) --error-logfile /var/log/gunicorn/error.log --log-level debug --workers 2 \
--bind unix:/var/www/server/Data_retrieving_ecommerce/proj.sock --timeout 300 wsgi:app

[Install]
WantedBy=multi-user.target
